16   fun mergeSort(A: IntArray): IntArray {
17       val n = A.size
18       if (n == 1)
19           return A
20       var C = A.slice(0 until (n / 2)).toIntArray()
21       var D = A.slice((n / 2) until n).toIntArray()
22   
23       C = mergeSort(C)
24       D = mergeSort(D)
25       return merge(C, D)
26   }
27   
28   fun merge(C: IntArray, D: IntArray): IntArray {
29       var i = 0
30       var j = 0
31       val B = IntArray(C.size + D.size)
32       for (k in B.indices) {
33           if (j == D.size) {
34               B[k] = C[i]
35               i += 1
36               continue
37           } else if (i == C.size) {
38               B[k] = D[j]
39               j += 1
40               continue
41           }
42   
43           if (C[i] < D[j]) {
44               B[k] = C[i]
45               i += 1
46           } else {
47               B[k] = D[j]
48               j += 1
49           }
50       }
51       return B
52   }
